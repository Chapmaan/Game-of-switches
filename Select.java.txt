import greenfoot.*;  // (World, Actor, GreenfootImage, Greenfoot and MouseInfo)
/**
 * Write a description of class Select here.
 * 
 * @author (your name) 
 * @version (a version number or a date)
 */
public class Select extends Actor
{
    //movement variable
    int deltaX;
    
    //also for the number of switches, but just for anti-cheat (Count = 3 to 8)
    int Count = 6;
    /**
     * Act - do whatever the Select wants to do. This method is called whenever
     * the 'Act' or 'Run' button gets pressed in the environment.
     */
    public void act() 
    {
        controls();
        restrictions();
    }
    
    public void restrictions()
    {
        //sets location
        setLocation(getX() + deltaX, getY() + 0);
        
        //loop in which determines where the selector can be (min X (55) and max X (xMax))
        for (int i = 1; i <= Count; i++)
        {
            int xMax = i * 65 + -10;
            if (Count == i)
            { 
                if (getX() < 55)        
                {
                   setLocation (55, 200);
                }
                if (getX() > xMax)
                {
                   setLocation (xMax, 200);
                }
            }
        }
    }   
    
    public void controls()
    {
       //stops movement
       deltaX = 0;
       //basic left/right movement
       if (Greenfoot.isKeyDown("left"))
       {
          deltaX = deltaX - 65;
       }
       if (Greenfoot.isKeyDown("right"))
       {
          deltaX = deltaX + 65;
       }
    }
}
